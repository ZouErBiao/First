1function [PP,PP_Cnt,PPP,PPP_Cnt] = Calculate_New_Indented_Boundary_Point(Point,P_Cnt)
 2%{
 3输入：
 4Point:内缩后的点集，格式为P_Cnt乘以2的矩阵
 5P_Cnt:点的数量
 6输出：
 7PP：各边的交点集，格式为n乘以7的矩阵
 8PPP：各边的顶点集，格式为n乘以7的矩阵
 9%}
10PP_Cnt = zeros(1,1);
11PPP_Cnt = zeros(1,1);
12PPP_Cnt(1,1) = P_Cnt;
13
14Pointss = zeros(P_Cnt,2);
15Pointss(1:P_Cnt,1:2) = flipud(Point(1:P_Cnt,1:2));%倒置变成逆时针
16Flag = zeros(1,1);
17PPP = zeros(100,7);
18%%此处把待处理的点集变成七列的矩阵
19for i=1:P_Cnt
20    PPP(i,1) = i;%原多边形顶点编号：1,2,3.......
21    PPP(i,2:3) = [0 0];%第二列放该点是否被访问标记0为没访问，1代表被访问，第三列为是否是交点标记，1为是交点，0为不是交点
22    PPP(i,4:5) = Pointss(i,:);%顶点坐标
23    PPP(i,6:7) = [0 0];%第六列存放交点的小序号边的起点，第七咧存放交点的大序号边的起点，其他为零
24end
25PP = zeros(50,7);%用于记录交点信息，8列矩阵
26k = zeros(1,1);
27%PP为交点集
28for i=3:P_Cnt
29    if i < P_Cnt
30        for j=1:i-2
31            Flag(1,1) = Judge_Intersect(Pointss(i,1),Pointss(i,2),Pointss(i+1,1),Pointss(i+1,2),Pointss(j,1),Pointss(j,2),Pointss(j+1,1),Pointss(j+1,2));%0为不相交，非零为相交
32            if Flag(1,1) ~= 0
33                k(1,1) = k(1,1) + 1;
34                PP(k(1,1),1:3) = [0,0,1];%序号位为零，访问标记位为0，是否交点位为1；
35                PP(k(1,1),4:5)  = CalculateLineCrossover(Pointss(i,1),Pointss(i,2),Pointss(i+1,1),Pointss(i+1,2),Pointss(j,1),Pointss(j,2),Pointss(j+1,1),Pointss(j+1,2));
36                PP(k(1,1),6:7) = [j,i];
37            end
38        end
39    else
40        for j=2:i-2
41            Flag(1,1) = Judge_Intersect(Pointss(i,1),Pointss(i,2),Pointss(1,1),Pointss(1,2),Pointss(j,1),Pointss(j,2),Pointss(j+1,1),Pointss(j+1,2));%0为不相交，非零为相交
42            if Flag(1,1) ~= 0
43                k(1,1) = k(1,1) + 1;
44                PP(k(1,1),1:3) = [0,0,1];%序号位为零，访问标记位为0，是否交点位为1；
45                PP(k(1,1),4:5)  = CalculateLineCrossover(Pointss(i,1),Pointss(i,2),Pointss(1,1),Pointss(1,2),Pointss(j,1),Pointss(j,2),Pointss(j+1,1),Pointss(j+1,2));
46                PP(k(1,1),6:7) = [j,i];
47            end
48        end
49    end 
50end
51PP_Cnt(1,1) = k(1,1);
