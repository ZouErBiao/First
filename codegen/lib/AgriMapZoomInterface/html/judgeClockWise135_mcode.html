1%% 判断地块是顺时针还是逆时针
 2function CWAFlag = judgeClockWise(lat_in,lon_in,len)
 3%输出:顺时针为1，逆时针为0
 4%算法说明：
 5% 对某一个点计算cross product = ((xi - xi-1),(yi - yi-1)) x ((xi+1 - xi),(yi+1 - yi)) = (xi - xi-1) * (yi+1 - yi) - (yi - yi-1) * (xi+1 - xi)
 6% 如果上式的值为正，逆时针；为负则是顺时针
 7% 而对于一般的简单多边形，则需对于多边形的每一个点计算上述值，如果正值比较多，是逆时针；负值较多则为顺时针。
 8% lat_in:x,lon_in:y
 9%%
10if 0
11    figure
12    plot(lat_in(1:len),lon_in(1:len),'*-');
13    hold on
14    plot(lat_in(1),lon_in(1),'r*','lineWidth',8);
15    plot(lat_in(2),lon_in(2),'*','lineWidth',8);
16end
17%% 判断
18CWAFlag = 0;
19if len < 3
20   return; 
21end
22%% 叉乘遍历计算
23count = 0;
24z = 0;
25for i=1:len
26    % 找到相邻两点
27    j = mod(i+1,len+1);
28    if j == 0
29        j = 1;%len;
30    end
31    k = mod(i+2,len+1);
32    if (i+2) > (len+1)
33        k = 2;
34    elseif k == 0
35        k = 1;%len;
36    end
37
38    % 计算叉乘
39    z = (lat_in(j) - lat_in(i)) * (lon_in(k) - lon_in(j));
40    z = z - (lon_in(j) - lon_in(i)) * (lat_in(k) - lat_in(j));
41    if z < 0
42        count = count - 1;
43    elseif z > 0
44        count = count + 1;
45    end
46end
47%
48if count >= 0
49    CWAFlag = 0;%逆时针
50else
51    CWAFlag = 1;%顺时针
52end
53%% Green 算法
54d = 0;
55for i=1:len-1
56    %d += - 0.5 * ( y[i + 1] + y[i]) * (x[i + 1] - x[i]);
57    d = d - 0.5 * (lon_in(i + 1) + lon_in(i)) * (lat_in(i + 1) - lat_in(i));
58end
59d = d - 0.5 * (lon_in(1) + lon_in(len)) * (lat_in(1) - lat_in(len));
60%
61if d >= 0
62    CWAFlag = 0;%逆时针
63else
64    CWAFlag = 1;%顺时针
65end
