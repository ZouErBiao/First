1function LPaiXu = Sort(L,Mm)
 2%对L进行从小到大排序,Index为排序前凹边两端点信息，Mm为L中元素的个数
 3LPaiXu = zeros(50,7);
 4for i =1:Mm,
 5    for j = i+1:Mm,
 6        if L(j,6) <= L(i,6)
 7            l = L(i,:);
 8            L(i,:) = L(j,:);
 9            L(j,:) = l;
10        end
11    end
12end
13LPaiXu(1:50,1:7) = L(1:50,1:7);
