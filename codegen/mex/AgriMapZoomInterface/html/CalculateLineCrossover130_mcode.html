1function LineCrossover = CalculateLineCrossover(x1,y1,x2,y2,x3,y3,x4,y4)
 2%计算两条相交线段之间的交点
 3LineCrossover = [0 0];
 4Middle = zeros(1,4);
 5if (x1~=x2 && y1==y2)&&(y2 == y3 && x3~=x4 && y3==y4)%两条线段都与x轴平行
 6    Middle = sort([x1 x2 x3 x4]);
 7    LineCrossover(1,1) = Middle(1,3);
 8    LineCrossover(1,2) = y1;
 9elseif (x1~=x2 && y1==y2) && (x3==x4 && y3~=y4 )%第一条鱼x轴平行，第二条与y轴平行
10    LineCrossover(1,1) = x3;
11    LineCrossover(1,2) = y1;
12elseif (x1~=x2 && y1==y2) && (x3~=x4 && y3~=y4)%第一条鱼x轴平行，第二条不与坐标轴平行
13    LineCrossover(1,1) = (y1-y3)*(x4-x3)/(y4-y3)+x3;
14    LineCrossover(1,2) = y1;
15elseif (x1 == x2 && y1~=y2) && (x3~=x4 && y3==y4)%第一条平行于y轴，第二条平行于x轴
16    LineCrossover(1,1) = x1;
17    LineCrossover(1,2) = y3;
18elseif (x1 == x2 && y1~=y2) && (x1 == x3 && x3 == x4 && y3~=y4)%第一条平行于y轴，第二条平行于y轴
19    LineCrossover(1,1) = x1;
20    Middle = sort([y1 y2 y3 y4]);
21    LineCrossover(1,2) = Middle(1,3);
22elseif (x1 == x2 && y1~=y2) && (x1~=x3 && y3~=y4)%第一条平行于y轴，第二条不与任何轴平行
23    LineCrossover(1,1) = x1;
24    LineCrossover(1,2) = (y4-y3)*(x1-x3)/(x4-x3)+y3;
25elseif (x1~=x2 && y1~=y2) && (x3~=x4 && y3==y4)%第一条不与任何轴平行，第二条与x轴平行
26    LineCrossover(1,1) = (y3-y1)*(x2-x1)/(y2-y1)+x1;
27    LineCrossover(1,2) = y3;
28elseif (x1~=x2 && y1~=y2) && (x3 == x4 && y3~=y4)
29    LineCrossover(1,1) = x3;
30    LineCrossover(1,2) = (y2-y1)*(x3-x1)/(x2-x1)+y1;
31else
32    a = (y4-y3)/(x4-x3);
33    b= (y2-y1)/(x2-x1);
34    if a ~= b %两线段线斜率不相同
35        LineCrossover(1,1) = (y1-y3 + a*x3-b*x1)/(a-b);
36        LineCrossover(1,2) = (a*b*x1-a*b*x3+b*y3-a*y1)/(b-a);
37    else%两线段斜率相同
38        Middle = sort([x1 x2 x3 x4]);
39        LineCrossover(1,1) = Middle(1,3);
40        LineCrossover(1,2) = b*(Middle(1,3)-x1) + y1;
41    end
42end
