1function Prep = Preprocess(Point,P_Cnt)
 2%对输入数据进行预处理，排除重合和连续三点在同一条直线上的情况
 3%{
 4输入：
 5%Point为输入的边界点
 6P_Cnt:边界点的个数
 7输出：
 8Prep为经过去重和去连续连续操作后的点集的结构体，格式为50乘以2
 9Prep.Preprocessing:格式为50乘以2
10Prep.Cnt:有效点的个数
11%}
12%%
13Prep.Preprocessing = zeros(100,2);
14Prep.Cnt = zeros(1,1);
15
16P_M = zeros(P_Cnt,3);
17Count = zeros(1,1);
18P_M(1:P_Cnt,1:2) = Point(1:P_Cnt,1:2);
19%P_M的第三列是标志位，为1的话则此行需删除
20Eps = zeros(1,1);
21Eps(1,1) = max([P_M(2,1)*0.005 P_M(2,2)*0.005]);%工程上要求的千分之五
22if all(abs(P_M(1,:)-P_M(2,:))<=Eps(1,1)) == 1
23    P_M(2,3) = 1;%为了占位给重复的地方替换为其他值，替换值要尽量不要跟数据采集点重复，采集点恰好与该值相等的概率极低（即：该值最好取地块的内部一点）
24    Count(1,1) = Count(1,1)+1;
25end
26for i=3:P_Cnt
27    for j=1:i-1
28        Eps(1,1) = max([P_M(i,1)*0.005 P_M(i,2)*0.005]);%工程上要求的千分之五
29        if all(abs(P_M(j,:)-P_M(i,:)) <= Eps(1,1)) == 1 
30            P_M(i,3) = 1;
31            Count(1,1) = Count(1,1)+1;
32            break;
33        end
34    end
35end
36%%
37%保留不重复的点
38Point_map = zeros(P_Cnt-Count(1,1),2);
39Cnt = zeros(1,1);%第一位用于下标计数
40
41for i =1:P_Cnt
42    if eq(P_M(i,3),0)
43        Cnt(1,1) = Cnt(1,1)+1;
44        Point_map(Cnt(1,1),1:2) = P_M(i,1:2);
45    end
46end
47%以上代码解决了点的重复问题
48%%
49%%以下代码解决连续三点在同一条直线上的问题
50NEdgevectorList = zeros(Cnt(1,1),2);
51NEdgevectorList(1:Cnt(1,1),1:2) = DPList(Point_map(1:Cnt(1,1),1:2),Cnt(1,1));
52NEdgevectorList(1:Cnt(1,1),1:2) = NormalizeDPList(NEdgevectorList(1:Cnt(1,1),1:2),Cnt(1,1));%各边的单位向量
53
54IndexPrepro = zeros(Cnt(1,1),1);%与前面单位向量相等的点的点号的存储,列向量
55Index_Cnt = zeros(1,1);
56if eq(NEdgevectorList(1,1),NEdgevectorList(Cnt(1,1),1)) && eq(NEdgevectorList(1,2),NEdgevectorList(Cnt(1,1),2))
57    IndexPrepro(1,1) = 1;%与前面单位向量相等的点号
58else
59    Index_Cnt(1,1) = Index_Cnt(1,1)+1;
60    IndexPrepro(1,1) = 0;
61end
62for i=2:Cnt(1,1)
63    if eq(NEdgevectorList(i,1),NEdgevectorList(i-1,1)) && eq(NEdgevectorList(i,2),NEdgevectorList(i-1,2))
64        IndexPrepro(i,1) = i;%与前面单位向量相等的点 号
65    else
66        Index_Cnt(1,1) = Index_Cnt(1,1)+1;
67        IndexPrepro(i,1) = 0;
68    end
69end
70
71Point_map1 = zeros(Index_Cnt(1,1),2);
72m = zeros(1,1);
73for i =1:Cnt(1,1)
74    if eq(IndexPrepro(i,1),0)
75        m(1,1) = m(1,1)+1;
76        Point_map1(m(1,1),1:2) = Point_map(i,1:2);
77    end
78end
79Prep.Preprocessing(1:Index_Cnt(1,1),1:2) =  Point_map1(1:Index_Cnt(1,1),1:2);
80Prep.Cnt(1,1) = Index_Cnt(1,1);
