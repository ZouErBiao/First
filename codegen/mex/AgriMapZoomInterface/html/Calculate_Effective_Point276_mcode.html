1function [YouXiao,YX_Cnt] = Calculate_Effective_Point(PointLast,N)
  2%{
  3%对经过内缩后的点集进行去无效环处理
  4输入：
  5PointLast为经内缩后的点集，为多行两列的矩阵,顺时针排列的%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  6输出：
  7YouXiao为经过处理后的保留下来的有效环的点集，为多行两列的矩阵
  8%}
  9YouXiao = zeros(100,2);
 10YX_Cnt = zeros(1,1);
 11
 12PP = zeros(50,7);
 13PPP = zeros(100,7);
 14PP_Cnt = zeros(1,1);
 15PPP_Cnt = zeros(1,1);
 16[PP(1:50,1:7),PP_Cnt(1,1),PPP(1:100,1:7),PPP_Cnt(1,1)] =Calculate_New_Indented_Boundary_Point(PointLast(1:N,1:2),N);%PPP为经过倒置、加列处理后的的七列顶点集，~为原顶点集倒置后的点集，PP为无序的交点集矩阵，也经过处理变成了七列的矩阵
 17PP(1:50,1:7) = Sort(PP(1:50,1:7),PP_Cnt(1,1));%对交点进行小序号边的起点进行排序，方便插入原顶点集
 18
 19PolygonList = zeros(100,7);
 20n = zeros(1,1);
 21[PolygonList(1:100,1:7),n(1,1)] = CalculateIntersection(PPP(1:PPP_Cnt(1,1),1:7),PPP_Cnt(1,1),PP(1:PP_Cnt(1,1),1:7),PP_Cnt(1,1));%中和交点的顶点集
 22%[PolygonList(1:100,1:7),n(1,1)] = Preprocessing(PolygonList(1:n(1,1),1:7),n(1,1));%去除交点和原顶点相同的点
 23
 24newPolygonnum = zeros(1,1);%经切割后，简单多边形的个数
 25b = zeros(50,1);
 26C_Cnt = zeros(1,4);%用于下标计数用
 27
 28ccc = zeros(n(1,1),1);
 29newPolygonList = zeros(n(1,1),7);
 30i=1;
 31tt = ones(1,1);
 32newPolygonnum(1,1) = newPolygonnum(1,1) +1;
 33newPolygonListPP = zeros(n(1,1),7);
 34
 35newPolygonListPP(i,:) = PolygonList(i,:);%每个新环的存储矩阵
 36C_Cnt(1,2) = 1;
 37PolygonList(1,2) = 1;
 38
 39row = zeros(n(1,1),1);%用于查找结果存储
 40m = zeros(1,4);
 41while tt(1,1) >= 1
 42    i=i+1;%%step4
 43    if i == n(1,1)+1
 44        i=1;
 45    end
 46        if PolygonList(i,3)==1 && (PolygonList(i,4) ~= newPolygonListPP(1,4) || PolygonList(i,5) ~= newPolygonListPP(1,5)) %判断是否为交点
 47            C_Cnt(1,1) = C_Cnt(1,1) + 1;
 48            b(C_Cnt(1,1),1) = i;
 49            
 50            C_Cnt(1,2) = C_Cnt(1,2) + 1;
 51            newPolygonListPP(C_Cnt(1,2),1:7) = PolygonList(i,1:7);%%以便此处添加一行
 52            
 53            row(1:n(1,1),1) = ismember(PolygonList(1:n(1,1),1:7),PolygonList(i,1:7),'rows');
 54            mm = find(row(1:n(1,1),1) == 1);
 55            [m(1,1),m(1,2)] = size(mm);
 56            if m(1,1)==1
 57                m(1,3) = PolygonList(i,7)+1;
 58                if m(1,3) > PPP_Cnt(1,1)%经过加一处理后，m(1,3)可能超过了未插入交点前的点集的点数
 59                    i = 0;
 60                else
 61                    for ii=1:n(1,1)
 62                        if eq(PolygonList(ii,1),m(1,3))
 63                            i = ii - 1;
 64                            break;
 65                        end
 66                    end
 67                    %m(1,4) = find(PolygonList(1:n(1,1),1)==m(1,3));
 68                    %i = m(1,4)-1;
 69                end
 70                
 71            else
 72                i = mm(2,1);
 73            end
 74        else%%Step6
 75            if PolygonList(i,4) ~= newPolygonListPP(1,4) || PolygonList(i,5) ~= newPolygonListPP(1,5) && PolygonList(i,2) == 0
 76                
 77                C_Cnt(1,2) = C_Cnt(1,2) + 1;
 78                newPolygonListPP(C_Cnt(1,2),:) = PolygonList(i,:);%%以便此处添加一行
 79                PolygonList(i,2) = 1;
 80            else
 81                PolygonList(i,2)=1;
 82                
 83                C_Cnt(1,3) = C_Cnt(1,3) + 1;
 84                ccc(C_Cnt(1,3),1) = C_Cnt(1,2);%存储每个环的顶点数
 85                
 86                newPolygonList(C_Cnt(1,4)+1:C_Cnt(1,4)+C_Cnt(1,2),1:7) = newPolygonListPP(1:C_Cnt(1,2),1:7);
 87                C_Cnt(1,4) = C_Cnt(1,4) + C_Cnt(1,2);
 88                
 89                newPolygonnum(1,1) = newPolygonnum(1,1) +1;%再次申请一个新的静态矩阵形式的空间
 90                newPolygonListPP = zeros(n(1,1),7);
 91                
 92                tttt=find(PolygonList(1:n(1,1),2)==0);%判断是否访问完了
 93                ttt = zeros(1,2);
 94                [ttt(1,1),ttt(1,2)]=size(tttt);
 95                if ttt(1,1) == 0
 96                    break;
 97                end
 98                newPolygonListPP(1,:) = PolygonList(b(newPolygonnum(1,1)-1,1),:);%重新开始一个环循环
 99                PolygonList(b(newPolygonnum(1,1)-1,1),2) = 1;
100                C_Cnt(1,2) = 1;
101                i = b(newPolygonnum(1,1)-1,1);
102            end 
103        end
104        tt(1,1) = tt(1,1)+1;
105        if tt(1,1) >= 100+PP_Cnt(1,1)*3
106            break;
107            %error('在处理无效环时陷入死循环！');
108        end
109end
110%%
111Aera = zeros(C_Cnt(1,3),3);
112C_Aera = zeros(1,1);
113FFLOG0 = zeros(1,1);
114ct = zeros(1,1);
115ind = zeros(1,2);
116if C_Cnt(1,3)==1
117    YouXiao(1:C_Cnt(1,4),1:2) = newPolygonList(1:C_Cnt(1,4),4:5);
118    YX_Cnt(1,1) = C_Cnt(1,4);
119else
120    for i=1:C_Cnt(1,3)
121        if i==1
122            FFLOG0(1,1) = judgeClockWise(newPolygonList(1:ccc(i),4),newPolygonList(1:ccc(i),5),ccc(i));
123            if FFLOG0(1,1) == 0
124                YouXiao(1:ccc(i),1:2) = newPolygonList(1:ccc(i),4:5);
125                C_Aera(1,1) = C_Aera(1,1) + 1;
126                Aera(C_Aera(1,1),1:3) = [Calculate_Area(YouXiao(1:ccc(i),1:2),ccc(i)),ccc(i),i];
127                %Mianji = [Mianji;[CalculateMianji(YouXiao,ccc(i)),ccc(i),i]];
128            end
129            %FFLOGzong = [FFLOGzong;FFLOG0];
130        else
131            ct(1,1) = sum(ccc(1:i-1));
132            FFLOG0(1,1) = judgeClockWise(newPolygonList(ct(1,1)+1:ct(1,1)+ccc(i),4),newPolygonList(ct(1,1)+1:ct(1,1)+ccc(i),5),ccc(i));
133            if FFLOG0(1,1)==0
134                YouXiao(1:ccc(i),1:2) = newPolygonList(ct(1,1)+1:ct(1,1)+ccc(i),4:5);
135                
136                C_Aera(1,1) = C_Aera(1,1) + 1;
137                Aera(C_Aera(1,1),1:3) = [Calculate_Area(YouXiao(1:ccc(i),1:2),ccc(i)),ccc(i),i];
138                %Mianji = [Mianji;[CalculateMianji(YouXiao,ccc(i)),ccc(i),i]];
139            end
140        end
141    end
142    YouXiao(1:100,1:2) = zeros(100,2);
143    [ind(1,1), ind(1,2)] = max(Aera(:,1)); %找到第一列最大值的序号
144    Aera(1,1:3) = Aera(ind(1,2),1:3); %显示最大值所在的一行的元素 
145    if Aera(1,3) == 1
146        YouXiao(1:Aera(1,2),1:2) = newPolygonList(1:Aera(1,2),4:5);
147        YX_Cnt(1,1) = Aera(1,2);
148    else
149        ct(1,1) = sum(ccc(1:Aera(1,3)-1,1));
150        YouXiao(1:Aera(1,2),1:2) = newPolygonList(ct(1,1)+1:ct(1,1)+Aera(1,2),4:5);
151        YX_Cnt(1,1) = Aera(1,2);
152    end
153end
