1function [Points,Cnt] = CalculateNewPoint(Point,P_Cnt,L,Theita)
  2%{
  3计算外扩后点的边界点
  4输入：
  5Point：需要外扩的多边形的边界点，格式为多行两列的矩阵，单位为m
  6L：需要外扩的距离，单位为m
  7Theita：是否为锐角的划分边界，单位为°
  8输出：
  9Points：经过外扩后的多边形的边界点，格式为多行两列的矩阵，单位为m
 10Cnt:输出点集有效点的个数
 11%}
 12%%
 13PreprocessingData = zeros(100,2);
 14PD_Cnt = zeros(1,1);
 15Prep = Preprocess(Point(1:P_Cnt,1:2),P_Cnt);%对数据进行除重和删除同一点操作；
 16PreprocessingData(1:100,1:2) = Prep.Preprocessing;
 17PD_Cnt(1,1) = Prep.Cnt;
 18
 19NEdgevectorList = zeros(100,2);
 20NEdgevectorList(1:PD_Cnt(1,1),1:2) = DPList(PreprocessingData(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));
 21NEdgevectorList(1:PD_Cnt(1,1),1:2) = NormalizeDPList(NEdgevectorList(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));%单位向量
 22Sina = zeros(1,PD_Cnt(1,1));
 23Cosa = zeros(1,PD_Cnt(1,1));
 24Sina(1,1:PD_Cnt(1,1)) = CalculateSina(NEdgevectorList(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));
 25Cosa(1,1:PD_Cnt(1,1)) = CalculateCosa(NEdgevectorList(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));
 26
 27P = zeros(100,2);
 28[P(1:100,1:2),PD_Cnt(1,1)] = Reduce_Angle(PreprocessingData(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1),L,Sina(1,1:PD_Cnt(1,1)),Cosa(1,1:PD_Cnt(1,1)),NEdgevectorList(1:PD_Cnt(1,1),1:2),Theita);
 29%%
 30PrepNew = Preprocess(P(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));
 31P(1:100,1:2) = zeros(100,2);
 32P(1:100,1:2) = PrepNew.Preprocessing;
 33PD_Cnt(1,1) = PrepNew.Cnt;
 34
 35Points = zeros(100,2);
 36Cnt = zeros(1,1);
 37n = zeros(1,1);
 38n(1,1) = PD_Cnt(1,1);
 39%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 40%{
 41P(PD_Cnt(1,1)+1,1:2) = P(1,1:2);
 42subplot(1,2,1);
 43plot(P(1:PD_Cnt(1,1)+1,1).',P(1:PD_Cnt(1,1)+1,2).','ro-');
 44title('未删除无效环');
 45xlabel('x');
 46ylabel('y');
 47grid;
 48%}
 49%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%10.9%%%%%%%%%%%%%%%%%%%%%%%%%%%
 50%[Points(1:100,1:2),Cnt(1,1)] = Calculate_Effective_Point(P(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));
 51%%
 52
 53%以下是保留边界点的循环
 54Flag = zeros(1,1);
 55tt_j = zeros(1,1);
 56ff = zeros(1,1);
 57tt = 1;
 58while tt ==1
 59    if n < 4
 60        Points(1:100,1:2) = P(1:100,1:2);
 61        Cnt(1,1) = PD_Cnt(1,1);
 62        break;
 63    end
 64    ff(1,1) = 0;
 65    for i=3:n
 66        Points(1:100,1:2) = zeros(100,2);
 67        if i < n
 68            for j=1:i-2
 69                Flag(1,1) = Judge_Intersect(P(i,1),P(i,2),P(i+1,1),P(i+1,2),P(j,1),P(j,2),P(j+1,1),P(j+1,2));%0为不相交，非零为相交
 70                if Flag(1,1) ~= 0
 71                    Points(1:j,1:2) = P(1:j,:);
 72                    Points(j+1,1:2) = CalculateLineCrossover(P(i,1),P(i,2),P(i+1,1),P(i+1,2),P(j,1),P(j,2),P(j+1,1),P(j+1,2));
 73                    Points(j+2:j+1+n-i,1:2) = P(i+1:n,:);
 74                    Cnt(1,1) = j+1+n-i;
 75                    ff(1,1) = 1;
 76                    if j == n-3
 77                        tt_j(1,1) = 1;
 78                    end
 79                    break;
 80                else
 81                    Points(1:j,1:2) = P(1:j,:);
 82                    Cnt(1,1) = j;
 83                end
 84                if j == n-3%i最多等于n-1,故j最多等于n-3
 85                    Points(Cnt(1,1)+1:Cnt(1,1)+3,1:2) = P(n-2:n,:);
 86                    Cnt(1,1) = Cnt(1,1)+3;
 87                    tt =0;
 88                end
 89            end
 90            if ff(1,1) == 1
 91                P(1:Cnt(1,1),1:2) = Points(1:Cnt(1,1),1:2);
 92                PD_Cnt(1,1) = Cnt(1,1);
 93                if tt_j(1,1) == 1
 94                    tt =0;
 95                end
 96                break;
 97            end
 98        else
 99            for j=2:i-2
100                Flag(1,1) = Judge_Intersect(P(i,1),P(i,2),P(1,1),P(1,2),P(j,1),P(j,2),P(j+1,1),P(j+1,2));%0为不相交，非零为相交
101                if Flag(1,1) ~= 0
102                    Points(1:j,1:2) = P(1:j,:);
103                    Points(j+1,1:2) = CalculateLineCrossover(P(i,1),P(i,2),P(i+1,1),P(i+1,2),P(j,1),P(j,2),P(j+1,1),P(j+1,2));
104                    %Points(j+2:j+1+n-i,1:2) = P(i+1:n,:);
105                    Cnt(1,1) = j+1;
106                    ff(1,1) = 1;
107                    if j == n-2
108                        tt_j(1,1) = 1;
109                    end
110                    break;
111                else
112                    Points(1:j,1:2) = P(1:j,:);
113                    Cnt(1,1) = j;
114                    if j == n-2%i最多等于n,故j最多等于n-2
115                        Points(Cnt(1,1)+1:Cnt(1,1)+2,1:2) = P(n-1:n,:);
116                        Cnt(1,1) = Cnt(1,1)+2;
117                        tt =0;
118                    end
119                end
120            end
121            if ff(1,1) == 1
122                P(1:Cnt(1,1),1:2) = Points(1:Cnt(1,1),1:2);
123                PD_Cnt(1,1) = Cnt(1,1);
124                if tt_j(1,1) == 1
125                    tt =0;
126                end
127                break;
128            end
129        end
130    end
131    PrepNew = Preprocess(P(1:PD_Cnt(1,1),1:2),PD_Cnt(1,1));
132    P(1:100,1:2) = PrepNew.Preprocessing;
133    PD_Cnt(1,1) = PrepNew.Cnt;
134    n(1,1) = PD_Cnt(1,1);
135end
