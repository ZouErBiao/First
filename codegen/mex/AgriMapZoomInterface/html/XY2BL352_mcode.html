1%L经度，B纬度
 2function [ B,L ] = XY2BL( x,y,L0 )
 3% 此函数用于进行高斯反算
 4
 5% CGCS2000椭球参数
 6% a=6378137;
 7% e=0.081819191042811;
 8% ee=0.0820944381518626;
 9% 1975国际椭球参数
10% a=6378140;
11% e=0.0818192214555235;
12
13%使用WGS-84椭球s
14a=6378137;
15b=6356752.3142;
16e=(sqrt(a^2-b^2))/a;
17% e1=(sqrt(a^2-b^2))/b; 
18
19ee=e/sqrt(1-e^2);
20% 计算底点的经度
21ap=1+3/4*e^2+45/64*e^4+175/256*e^6+11025/16384*e^8+43659/65536*e^10;
22bp=3/4*e^2+15/16*e^4+525/512*e^6+2205/2048*e^8+72765/65536*e^10;
23cp=15/64*e^4+105/256*e^6+2205/4096*e^8+10395/16384*e^10;
24dp=35/512*e^6+315/2048*e^8+31185/131072*e^10;
25ep=315/16384*e^8+3465/65536*e^10;
26fp=639/131072*e^10;
27
28B=x/(a*(1-e^2)*ap);
29deltaB=1000;
30while deltaB>10^(-20)
31    Bi=1/ap*(x/(a*(1-e^2))+bp/2*sin(2*B)-cp/4*sin(4*B)+dp/6*sin(6*B)-ep/8*sin(8*B)+fp/10*sin(10*B));
32    deltaB=abs(Bi-B);
33    B=Bi;
34end
35Bf=B;
36
37% 计算大地经纬度BL
38tf=tan(Bf);
39Vf=sqrt(1+ee^2*cos(Bf)*cos(Bf));
40Wf=sqrt(1-e^2*sin(Bf)*sin(Bf));
41Nf=a/Wf;
42etaf=ee*cos(Bf);
43B=Bf-1/2*Vf^2*tf*((y/Nf)*(y/Nf)-1/12*(5+3*tf^2+etaf^2-9*etaf^2*tf^2)*(y/Nf)^4+1/360*(61+90*tf^2+45*tf^4)*(y/Nf)^6);
44LP=1/cos(Bf)*((y/Nf)-1/6*(1+2*tf^2+etaf^2)*(y/Nf)^3+1/120*(5+28*tf^2+24*tf^4+6*etaf^2+8*etaf^2*tf^2)*(y/Nf)^5);
45B=B/pi*180;
46LP=LP/pi*180;
47L=L0+LP;
48
49end
50
