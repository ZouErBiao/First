1function [P,P_Cnt] = Reduce_Angle(Point,n,L,Sina,Cosa,NEdgevectorList,Theita)
 2%对点Point进行去锐角化
 3%{
 4输入：
 5Point:需要锐角化的边界点集，格式为n乘以2
 6n：Point中点的个数
 7L：外扩的距离单位米
 8Sina：个顶点出的角的sin值
 9Cosa：个顶点出的角的cosa值
10NEdgevectorList:各边的单位向量
11Theita：锐角临界值
12输出：
13P：经去锐角化的点集，格式为P_Cnt乘以2
14P_Cnt:经处理后的点集中点的个数
15%}
16P = zeros(100,2);
17P_Cnt = zeros(1,1);
18
19k = 0;
20p0 = zeros(1,2);
21a1 = zeros(1,1); 
22a2 = zeros(1,1); 
23if abs(Sina(1,1)) > sin(Theita*3.1415926/180)%0.3420
24    P(1,1:2) = Point(1,1:2) - (NEdgevectorList(1,1:2) - NEdgevectorList(n,1:2)) * L / Sina(1,1);
25    P_Cnt(1,1) = P_Cnt(1,1) + 1;
26elseif abs(Sina(1,1)) <= sin(Theita*3.1415926/180)
27    if Sina(1,1) > 0 && Cosa(1,1) < 0
28        k= k+1;
29        p0(1,1:2) = -NEdgevectorList(1,1:2) + NEdgevectorList(n,1:2);
30        a1(1,1) = p0(1,1) / sqrt(p0(1,1)*p0(1,1)+p0(1,2)*p0(1,2));
31        a2(1,1) = p0(1,2) / sqrt(p0(1,1)*p0(1,1)+p0(1,2)*p0(1,2));
32        P(1+2*k-2,1) = Point(1,1) + L*(a1(1,1)*sqrt((1 + Cosa(1,1))/2) - a2(1,1)*sqrt((1 - Cosa(1,1))/2));
33        P(1+2*k-2,2) = Point(1,2) + L*(a2(1,1)*sqrt((1 + Cosa(1,1))/2) + a1(1,1)*sqrt((1-Cosa(1,1))/2));
34        
35        P(1+2*k-1,1:2) = Point(1,1:2) + [a1(1,1) a2(1,1)]* L;
36         
37        P(1+2*k,1) = Point(1,1) + L*(a1(1,1)*sqrt((1 + Cosa(1,1))/2) + a2(1,1)*sqrt((1-Cosa(1,1))/2));
38        P(1+2*k,2) = Point(1,2) + L*(a2(1,1)*sqrt((1 + Cosa(1,1))/2) - a1(1,1)*sqrt((1-Cosa(1,1))/2));
39        P_Cnt(1,1) = P_Cnt(1,1) + 3;
40    else
41         P(1,1:2) = Point(1,1:2) - (NEdgevectorList(1,1:2) - NEdgevectorList(n,1:2)) * L / Sina(1,1);
42         P_Cnt(1,1) = P_Cnt(1,1) + 1;
43    end
44end
45for i=2:n
46    if abs(Sina(1,i)) > sin(Theita*3.1415926/180) 
47        P(i+2*k,:) = Point(i,:) - (NEdgevectorList(i,:) - NEdgevectorList(i-1,:)) * L / Sina(1,i);
48        P_Cnt(1,1) = P_Cnt(1,1) + 1;
49    elseif abs(Sina(1,i)) <= sin(Theita*3.1415926/180)
50        if Sina(1,i) > 0 && Cosa(1,i) < 0
51            k = k+1;
52            p0(1,1:2) = -NEdgevectorList(i,1:2) + NEdgevectorList(i-1,1:2);
53            a1(1,1) = p0(1,1) / sqrt(p0(1,1)*p0(1,1)+p0(1,2)*p0(1,2));
54            a2(1,1) = p0(1,2) / sqrt(p0(1,1)*p0(1,1)+p0(1,2)*p0(1,2));
55            P(i+2*k-2,1) = Point(i,1) + L*(a1(1,1)*sqrt((1 + Cosa(1,i))/2) - a2(1,1)*sqrt((1 - Cosa(1,i))/2));
56            P(i+2*k-2,2) = Point(i,2) + L*(a2(1,1)*sqrt((1 + Cosa(1,i))/2) + a1(1,1)*sqrt((1 - Cosa(1,i))/2));
57            
58            P(i+2*k-1,:) = Point(i,:) + [a1(1,1) a2(1,1)] * L;
59            
60            P(i+2*k,1) = Point(i,1) + L*(a1(1,1)*sqrt((1 + Cosa(1,i))/2) + a2(1,1)*sqrt((1 - Cosa(1,i))/2));
61            P(i+2*k,2) = Point(i,2) + L*(a2(1,1)*sqrt((1 + Cosa(1,i))/2) - a1(1,1)*sqrt((1 - Cosa(1,i))/2));  
62           P_Cnt(1,1) = P_Cnt(1,1) + 3;
63        else
64            P(i+2*k,1:2) = Point(i,1:2) - (NEdgevectorList(i,1:2) - NEdgevectorList(i-1,1:2)) * L / Sina(1,i);
65            P_Cnt(1,1) = P_Cnt(1,1) + 1;
66        end
67    end
68end
